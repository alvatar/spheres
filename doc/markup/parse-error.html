(define parser-error
  (lambda  args
    (if
      (port? (car args))
      (cerr nl "Error at position " 
	    (file-position (car args)) nl
	    (cdr args))
      (cerr nl "Error in error handler: its first parameter is not a port" 
	    nl args))
    (cerr nl)
    ; (exit -1)  ; this exit makes me completely insane!
    (raise -1)
))
